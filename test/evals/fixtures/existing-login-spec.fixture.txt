import { expect, test } from '../../fixtures'
import { LoginPage } from '../../pages'

test.describe('Login Page', () => {
  let loginPage: LoginPage

  test.beforeEach(async ({ page }) => {
    loginPage = new LoginPage(page)
    await loginPage.goto()
  })

  // biome-ignore lint/correctness/noUnusedFunctionParameters: fixture uses page via loginPage
  test('should display login form', async ({ page }) => {
    await expect(loginPage.emailInput).toBeVisible()
    await expect(loginPage.passwordInput).toBeVisible()
    await expect(loginPage.signInButton).toBeVisible()
  })

  test('should login with valid credentials', async ({ page }) => {
    await loginPage.emailInput.fill('test.user@example.com')
    await loginPage.passwordInput.fill('P@ssw0rd123!')
    // BUG: stale selector — should be data-testid="signin-btn"
    await page.locator('[data-testid="login-btn"]').click()
    // BUG: deprecated API — should use waitForURL
    await page.waitForNavigation()
    await expect(page).toHaveURL('/dashboard')
  })

  test('should show error for invalid credentials', async ({ page }) => {
    await loginPage.emailInput.fill('test.user@example.com')
    await loginPage.passwordInput.fill('WrongPassword!')
    await page.locator('[data-testid="login-btn"]').click()
    // BUG: hardcoded wait — should use proper assertion wait
    await page.waitForTimeout(2000)
    await expect(loginPage.errorMessage).toBeVisible()
    await expect(loginPage.errorMessage).toHaveText('Invalid email or password')
  })

  test('should validate empty email field', async ({ page }) => {
    await loginPage.passwordInput.fill('P@ssw0rd123!')
    await page.locator('[data-testid="login-btn"]').click()
    await page.waitForTimeout(2000)
    await expect(page.locator('.email-error')).toHaveText('Email is required')
  })

  test('should validate empty password field', async ({ page }) => {
    await loginPage.emailInput.fill('test.user@example.com')
    await page.locator('[data-testid="login-btn"]').click()
    await expect(page.locator('.password-error')).toHaveText('Password is required')
  })

  test('should toggle remember me checkbox', async ({ page }) => {
    const rememberMe = page.getByRole('checkbox', { name: 'Remember Me' })
    await expect(rememberMe).not.toBeChecked()
    await rememberMe.check()
    await expect(rememberMe).toBeChecked()
  })

  test('should navigate to forgot password page', async ({ page }) => {
    await page.getByRole('link', { name: 'Forgot Password?' }).click()
    await page.waitForNavigation()
    await expect(page).toHaveURL('/forgot-password')
  })

  // biome-ignore lint/correctness/noUnusedFunctionParameters: fixture uses page via loginPage
  test('should mask password input', async ({ page }) => {
    await expect(loginPage.passwordInput).toHaveAttribute('type', 'password')
    await loginPage.passwordInput.fill('P@ssw0rd123!')
    await expect(loginPage.passwordInput).toHaveAttribute('type', 'password')
  })
})
